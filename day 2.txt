class Solution {

    public long minTime(long a[], long b[], long n) {
        // Your code goes here 
        long sum1=0,sum2=0;
        for(int i=0;i<n;i++){
            if(i%2==0){
                sum1+=a[i];
                sum2+=b[i];
            }
            else{
                sum1+=b[i];
                sum2+=a[i];
            }
        }
        long min=sum1;
        min=(sum1<sum2? sum1:sum2);
        return min;
    }
}
